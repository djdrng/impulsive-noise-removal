def apply_impulsive_noise(image_data: np.ndarray, amplitude, probability):
  for i in range(len(image_data)):
    for j in range(len(image_data[0])):
      if (np.random.random() < probability):
        for k in range(len(image_data[0][0])):
            image_data[i][j][k] = (image_data[i][j][k] + amplitude) % 255
  return image_data

noisy_data = apply_impulsive_noise(data, 100, 0.2)
noisy_image = Image.fromarray(noisy_data)
noisy_image.show()
